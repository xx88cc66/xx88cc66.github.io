        function parseJSONFromResponse(input) {
            const jsonStart = input.indexOf('{');
            if (jsonStart === -1) {
                throw new Error('未找到有效的JSON部分');
            }
            const jsonString = input.slice(jsonStart);
            return JSON.parse(jsonString);
        }

        function extractDomain() {
            handleExtraction(item => item.domain, '未找到域名数据。');
        }

        function extractServiceName() {
            handleExtraction(item => item.serviceName, '未找到App或小程序数据。');

        function handleExtraction(mapper, emptyMessage) {
            const jsonInput = document.getElementById('jsonInput').value;
            const errorMessage = document.getElementById('errorMessage');
            const resultList = document.getElementById('resultList');
            errorMessage.textContent = '';
            resultList.innerHTML = '';

            try {
                const response = parseJSONFromResponse(jsonInput);
                const results = response.params.list.map(mapper).filter(Boolean);

                if (results.length === 0) {
                    resultList.innerHTML = `<li>${emptyMessage}</li>`;
                } else {
                    results.forEach(result => {
                        const li = document.createElement('li');
                        li.textContent = result;
                        resultList.appendChild(li);
                    });
                }
            } catch (error) {
                errorMessage.textContent = '无效的HTTP响应内容或JSON格式，请检查输入。';
            }
        }

        function copyResults() {
            const resultList = document.getElementById('resultList');
            const items = Array.from(resultList.querySelectorAll('li')).map(li => li.textContent);
            if (items.length > 0) {
                navigator.clipboard.writeText(items.join('\n')).then(() => {
                    alert('结果已复制到剪贴板！');
                });
            } else {
                alert('没有可复制的内容，请先提取数据。');
            }
        }
